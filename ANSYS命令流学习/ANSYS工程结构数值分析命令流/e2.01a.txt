!EX2.1A-整圈数圆柱形螺旋弹簧的几何建模
FINISH
/CLEAR
/PREP7
!1.定义弹簧参数
D=4
C=8
N=10
DZ=C*D
T=DZ/2.5
*IF,T,LT,D,THEN
T=D
*ENDIF
!2.创建一圈螺旋线
CSYS,1
K,1,DZ/2,0,-T/2
K,2,DZ/2,180
L,1,2
CSYS,0
LSYMM,Z,1
LSYMM,Y,2,,,,,1
NUMMRG,ALL
CM,L1,LINE
!3.在螺旋线端部创建簧丝截面
KWPAVE,1
WPROTA,,90
CYL4,,,D/2
!4.沿L1路径拖拉圆面创建体、复制体等
VDRAG,1,,,,,,L1
VGEN,N,ALL,,,,,T
NUMMRG,KP
WPCSYS
